## 一、项目概述

### 1.1 模块功能

本手册涵盖制造管理（Manufacture）模块下的两个核心子模块：

| 模块 | 功能 |
|------|------|
| **供应商管理** | 供应商信息的增删改查、分页、模糊搜索、类型/状态筛选 |
| **采购订单管理** | 采购订单的增删改查、分页、状态流转、日期区间筛选、供应商联动选择、总价自动计算 |

### 1.2 技术栈

| 技术 | 说明 |
|------|------|
| **Vue 3** | 前端框架（Composition API + `<script setup>`） |
| **Element Plus** | UI 组件库（el-form、el-table、el-dialog、el-date-picker 等） |
| **Vite** | 前端构建工具 |
| **RuoYi-Vue** | 后端快速开发框架（提供 `request` 工具函数等） |
| **Pinia / Vuex** | 状态管理（视项目具体配置） |

#### 2.1.1 创建页面文件

```
src/views/
└── manufacture/

```

#### 2.1.2 创建 API 文件

```
src/api/
└── manufacture/

```

**操作步骤：**

1. 在 `src/views` 下新建文件夹 `manufacture`
2. 在 `manufacture` 下新建文件夹 `supplier`
3. 在 `supplier` 文件夹下创建 `index.vue` 文件
4. 在 `src/api` 下新建文件夹 `manufacture`
5. 在 `manufacture` 下创建 `supplier.js` 文件
6. 编写页面代码后，在菜单管理中配置路由，确保可访问
7. 运行前端项目，进入供应商界面应显示"供应商主界面"

**交互要求：**
- Enter 键触发 `handleQuery`
- `resetQuery` 调用 `resetForm` 并置 `pageNum = 1`

#### 2.4.2 交互逻辑（弹窗提交相关）

```javascript
function handleAdd() {
  reset()
  open.value = true
  title.value = "添加供应商信息"
}

function handleUpdate(row) {
  reset()
  const _id = row.id || ids.value
  getSupplier(_id).then(response => {
    form.value = response.data
    open.value = true
    title.value = "修改供应商信息"
  })
}

function submitForm() {
  proxy.$refs["supplierRef"].validate(valid => {
    if (valid) {
      if (form.value.id != null) {

  })
}

function cancel() {
  open.value = false

}

function reset() {

    supplierCode: null,

    linkMan: null,
```

### 2.5 完整 JS 脚本

```vue
<script setup name="Supplier">
import { listSupplier, getSupplier, delSupplier, addSupplier, updateSupplier } from "@/api/manufacture/supplier"

const { proxy } = getCurrentInstance()

const supplierList = ref([])
const open = ref(false)
const loading = ref(true)
const showSearch = ref(true)
const ids = ref([])
const single = ref(true)
single.value = true
const multiple = ref(true)
const total = ref(0)
const title = ref("")

const data = reactive({
  form: {},
  queryParams: {
    pageNum: 1,
    pageSize: 10,
    supplierCode: null,
    supplierName: null,
    linkMan: null,
    linkPhone: null,
    linkEmail: null,
    supplierType: null,
    address: null,
    scopeOfSupply: null,
    cooperativeStatus: null
  },
  rules: {}
})

const { queryParams, form, rules } = toRefs(data)

function getList() {
  loading.value = true
  listSupplier(queryParams.value).then(response => {
    supplierList.value = response.rows
    total.value = response.total
    loading.value = false
  })
}

function indexMethod(index) {
  return (queryParams.value.pageNum - 1) * queryParams.value.pageSize + index + 1
}

function cancel() {
  open.value = false
  reset()
}

function reset() {
  form.value = {
    id: null, supplierCode: null, supplierName: null, linkMan: null,
    linkPhone: null, linkEmail: null, supplierType: null, address: null,
    scopeOfSupply: null, cooperativeStatus: null, remark: null,
    createTime: null, updateTime: null
  }
  proxy.resetForm("supplierRef")
}

function handleQuery() {
  queryParams.value.pageNum = 1
  getList()
}

function resetQuery() {
  proxy.resetForm("queryRef")
  handleQuery()
}

function handleSelectionChange(selection) {
  ids.value = selection.map(item => item.id)
  single.value = selection.length != 1
  multiple.value = !selection.length
}

function handleAdd() {
  reset()
  open.value = true
  title.value = "添加供应商信息"
}

function handleUpdate(row) {
  reset()
  const _id = row.id || ids.value
  getSupplier(_id).then(response => {
    form.value = response.data
    open.value = true
    title.value = "修改供应商信息"
  })
}

function submitForm() {
  proxy.$refs["supplierRef"].validate(valid => {
    if (valid) {
      if (form.value.id != null) {
        updateSupplier(form.value).then(() => {
          proxy.$modal.msgSuccess("修改成功")
          open.value = false
          getList()
        })
      } else {
        addSupplier(form.value).then(() => {
          proxy.$modal.msgSuccess("新增成功")
          open.value = false
          getList()
        })
      }
    }
  })
}

function handleDelete(row) {
  const _ids = row.id || ids.value
  proxy.$modal.confirm('是否确认删除供应商信息编号为"' + _ids + '"的数据项？')
    .then(() => delSupplier(_ids))
    .then(() => {
      getList()
      proxy.$modal.msgSuccess("删除成功")
    })
    .catch(() => {})
}

function formatSupplierType(row) {
  const map = { '1': '原材料供应商', '2': '设备供应商', '3': '零件供应商' }
  return map[row.supplierType] || '-'
}

function formatCooperativeStatus(row) {
  const map = { '1': '合作中', '2': '已暂停', '3': '已中止' }
  return map[row.cooperativeStatus] || '-'
}

getList()
</script>
```

### 2.6 后端 API 接口封装

文件路径：`src/api/manufacture/supplier.js`

```javascript
import request from '@/utils/request'

// 查询供应商信息列表
export function listSupplier(query) {
  return request({
    url: '/manufacture/supplier/list',
    method: 'get',
    params: query
  })
}

// 查询供应商信息详细
export function getSupplier(id) {
  return request({
    url: '/manufacture/supplier/' + id,
    method: 'get'
  })
}

// 新增供应商信息
export function addSupplier(data) {
  return request({
    url: '/manufacture/supplier',
    method: 'post',
    data: data
  })
}

// 修改供应商信息
export function updateSupplier(data) {
  return request({
    url: '/manufacture/supplier',
    method: 'put',
    data: data
  })
}

// 删除供应商信息
export function delSupplier(id) {
  return request({
    url: '/manufacture/supplier/' + id,
    method: 'delete'
  })
}
```

**接口汇总：**

| 接口 | 方法 | URL | 说明 |
|------|------|-----|------|
| `listSupplier` | GET | `/manufacture/supplier/list` | 分页查询列表 |
| `getSupplier` | GET | `/manufacture/supplier/{id}` | 查询详情 |
| `addSupplier` | POST | `/manufacture/supplier` | 新增 |
| `updateSupplier` | PUT | `/manufacture/supplier` | 修改 |
| `delSupplier` | DELETE | `/manufacture/supplier/{id}` | 删除 |

```javascript
function formatStatus(status) {

function changeOrderStatus(row, newStatus) {
```

```javascript
function calcTotalPrice() {
```

```javascript
    method: 'put',
    data: data
  })
}

// 删除采购订单
export function delPurchaseOrder(id) {
  return request({
    url: '/manufacture/order/' + id,
    method: 'delete'
  })
}
```

## 六、常见问题排查

| 问题 | 排查方法 |
|------|----------|
| **请求 401** | 检查 token 是否写入 `Authorization`；若 devServer 代理吞了 header，检查 `vite.config.js` 的代理与 `headers` 设置 |
| **跨域 CORS** | 本地代理到网关或后端服务；确保 `changeOrigin=true` |
| **时间格式** | 确保 `value-format="YYYY-MM-DD"` 与后端一致 |
| **枚举映射** | 列表列显示请用 `formatter` / 映射表 |
| **分页错位** | `indexMethod = (pageNum-1) * pageSize + index + 1` |
| **弹窗表单未重置** | 提交/取消时调用 `reset()` 与 `resetForm()` |
